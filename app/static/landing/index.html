# Document AI Pipeline: PDF → OCR → Chat Summary
import requests

# 1. Extract text from PDF
pdf_result = requests.post("https://cerebrum-blocks.onrender.com/v1/execute", json={
    "block": "pdf",
    "input": "contract.pdf",
    "params": {"extract": "text"}
}).json()

# 2. Get AI summary via Chat (DeepSeek = cheapest!)
chat_result = requests.post("https://cerebrum-blocks.onrender.com/v1/chat", json={
    "message": f"Summarize: {pdf_result['result']['text'][:2000]}",
    "model": "deepseek-chat",
    "provider": "deepseek"
}).json()

print(chat_result["text"])
